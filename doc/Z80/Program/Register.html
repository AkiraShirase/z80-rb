def initialize(name, opc)
        @name = name
        @index = 0
        @opc = opc & 0xff
        @bit8 = name.size != 2 and opc & 7 == opc
        @prefix = name.index(xx) ? "\xDD" : name.index(yy) ? "\xFD" : nil
end
